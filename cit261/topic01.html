Loops Example 1: for

Code:

var output = "";
var x, y;

for (x = y = 0; x < 10; x++) {
     y += x;
     output += "x is [" + x + "], y is [" + y + "]<br>";
}
output += "<br>Final Value for y is [" + y + "]<br>";

Loops Example 4: while

Code:

var output = "";
var x, y;
x = y = 0;

while (x < 10) {
     y += x;
     output += "x is [" + x + "], y is [" + y + "]<br>";
     x++;
}

output += "<br>Final Value for y is [" + y + "]<br>";

Loops Example 5: do while

Code:

var output = "";
var x, y;
x = y = 0;

do {
     y += x;
     output += "x is [" + x + "], y is [" + y + "]<br>";
     x++;
} while (x < 10);

output += "<br>Final Value for y is [" + y + "]<br>";
