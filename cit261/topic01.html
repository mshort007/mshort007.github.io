Loops Example 1: for

Code:

var output = "";
var x, y;

for (x = y = 0; x < 10; x++) {
     y += x;
     output += "x is [" + x + "], y is [" + y + "]<br>";
}
output += "<br>Final Value for y is [" + y + "]<br>";

Code:

}
if (num < 5) {
     output += "Random Number is Smaller than 5!<br>";

Code:

     case 0:
     case 1:
     case 2:
